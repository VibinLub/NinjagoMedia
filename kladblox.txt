    let dropdowns = document.getElementsByClassName('dropdown');

    for (let i = 0; i < dropdowns.length; i++) {
        dropdowns[i].addEventListener('mouseover', function() {
            this.getElementsByClassName('dropdown-content')[0].style.display = 'block';
        });
        dropdowns[i].addEventListener('mouseout', function() {
            this.getElementsByClassName('dropdown-content')[0].style.display = 'none';
        });
    }
});

// Function to open modal with video
function openModal(title, videoUrl) {
    const overlay = document.querySelector('.overlay');
    overlay.style.display = 'flex';

    const modal = document.createElement('div');
    modal.classList.add('modal');

    modal.innerHTML = `
        <span class="close-btn">&times;</span>
        <h2>${title}</h2>
        <iframe width="100%" height="315" src="${videoUrl}" frameborder="0" allowfullscreen></iframe>
    `;

    modal.querySelector('.close-btn').addEventListener('click', () => closeModal(modal, overlay));

    overlay.appendChild(modal);
}

// Function to close modal
function closeModal(modal, overlay) {
    overlay.style.display = 'none';
    modal.remove();
}

    populateSpecials();
